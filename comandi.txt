PULIZIA TOTALE MINIKUBE (cancella anche coreDNS ATTENZIONE!!!)

kubectl delete deployments,services,configmaps,persistentvolumeclaims,secrets -n default --all

CONVERSIONE E CREAZIONE PODS

minikube addons enable storage-provisioner
minikube addons enable default-storageclass

(Se gli addons erano disattivati)
minikube stop
minikube start

mkdir kompose-files

# per creare i pv
kubectl apply -f ./kompose-files/pv

kubectl apply -f ./kompose-files

BUILDARE L'IMMAGINE CKAN

docker build -t ckan-unime:latest ./ckan
minikube image load ckan-unime:latest
